Install-WindowsFeature DHCP -IncludeManagementTools
Get-NetAdapter
New-NetIPAddress -InterfaceAlias ACA VA EL ADAPTADOR DE ETHERNET QUE VAS A USAR EL COMANDO ANTERIOR MUESTRA LOS DISPONIBLES -IPAddress ACA VA EL IP ESTATICO DEL ADAPTADOR AKA EL SERVER -PrefixLength 24 -DefaultGateway EL DEFAULT GATEWAY YO PONGO EL PRIMER IP DE ESA FAMILIA EJ SI PUSISTE EL 192.162.100.10 EL DEFAULT GATEWAY ES EL 192.168.100.1
Set-DnsClientServerAddress -InterfaceAlias MISMO QUE ARRIBA EL ALIAS -ServerAddresses 8.8.8.8,8.8.4.4
Add-DhcpServerv4Scope -Name NOMBRE DEL SERVER O DEL SCOPE OSEA DEL DHCP -StartRange ACA VA EL RANGO DEL CUAL EMPIEZA A ASIGNAR IP Y -EndRange Y ACA EN DONDE TERMINA -SubnetMask NETMASK SIEMPRE QUE USO LOS OCTETOS DE NORMAL USO 255.255.255.0 -State Active
Set-DhcpServerv4OptionValue -ScopeId ACA VA LA DIRECCION RAIZ DEL IP -OptionId 3 -Value ACA VA LA MISMA DEFAULT GATEWAY QUE PUSISTE AL ADAPTADOR
Set-DhcpServerv4OptionValue -ScopeId IGUAL LA DIRECCION RAIZ -OptionId 6 -Value  8.8.8.8,8.8.4.4(ACA PUEDES USAR LAS QUE QUIERAS ESAS SON LAS DEL GUGUL)
Get-Service DHCPServer ESTE COMANDO CHECA SI EL DHCP ESTA CORRIENDOSE
Get-DhcpServerv4Scope SI NO LA CAGASTE EN NADA QUE ES DIFICIL CON LOS COMANDOS QUE TE DI TAMBIEN TE DIGO ESTE COMANDO DEBE MOSTRAR LOS V4CSOPES(SERVERS DHCP CONFIGURADOS)
